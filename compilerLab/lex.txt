import java.io.IOException;
import java.util.Vector;
public class Operator_precedence {
    public static void main(String[] args) throws IOException {
        Vector<Integer>v=new Vector<>(10);
        for (int i = 0; i <8 ; i++) {
            v.add(i,i*10);
        }
        System.out.println(v.lastIndexOf(v));

    }
}
